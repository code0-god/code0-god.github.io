## C++ Highlighting Test
Below Code includes every token class samples.

```cpp
// Single-line comment (Comment.Single)
/* Multi-line comment
    (Comment.Multiline) */
#define MACRO_VAL 42        // Preprocessor directive (Keyword.Pseudo, Comment)
#define MACRO_FUNC(x, y)  ((x) + (y))

#include <vector>          // Name.Builtin, Punctuation
namespace zerogod {
    template <typename T>      // Keyword.Declaration, Name.Builtin
    class MyClass {            // Name.Class
    public:
        MyClass(T v) : value(v) {}  // Name.Function, Name.Variable
        T getValue() const {        // Keyword.Constant
            return value;           // Keyword
        }
    private:
        T value;                // Name.Variable
    };
}
int main() {               // Keyword.Reserved
    std::vector<int> v = vector<int>(0, 5);  // Literal.Number, Punctuation
    auto sum = 0;          // Keyword.Type, Name.Variable
    for (auto x : v) {     // Keyword, Name.Variable
        sum += x;          // Operator
    }
    std::string s = "Test string\n";  // Literal.String
    std::cout << s           // Name.Builtin, Operator
              << "Sum: "     // Literal.String
              << sum         // Name.Variable
              << std::endl;  // Name.Builtin

    int error = ?;          // intended err token
    return 0;               // Keyword
}
```
