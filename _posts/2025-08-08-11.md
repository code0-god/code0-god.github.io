{% code_runner id="sum-1" language="cpp" filename="sum" %}
#include <cstdio>
int main() {
  int a = 2, b = 3;
  printf("2 + 3 = %d\n", a + b);
  return 0;
}
{% endcode_runner %}

{% code_runner id="loop-1" language="cpp" filename="loop" %}
#include <cstdio>
int main() {
  for (int i = 1; i <= 100; ++i) {
    printf("Line %d\n", i);
  }
  return 0;
}
{% endcode_runner %}
